first = input("What would you like to become a palindrome?")
print("Input:", first)
length = len(first)
firstpart = first[0:(length-1)]
lastchar = first[-1]
flip = first[::-1]

if first == flip:
    print("Is a palindrome")
else:
    print("Isn't a palindrome")
    print("Palindrome:",firstpart+flip)
